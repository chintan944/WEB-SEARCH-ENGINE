It can be added easily to columns and rows using the POI API's methods.Lets see an example in which we are applying outlining.Apache POI Outlining Examplepackage poiexample;
import java.io.FileOutputStream;
import java.io.OutputStream;
import org.apache.poi.hssf.usermodel.HSSFWorkbook;
import org.apache.poi.ss.usermodel.Sheet;
import org.apache.poi.ss.usermodel.Workbook;
public class OutlineExample {
	public static void main(String[] args) {
		 try (OutputStream fileOut = new FileOutputStream("Javatpoint.xls")) {
			 Workbook wb = new HSSFWorkbook();
			 Sheet sheet1 = wb.createSheet("Sheet");
			 sheet1.groupRow( 4, 10 );
			 sheet1.groupColumn( 2, 8 );
		     wb.write(fileOut);
		 }catch(Exception e) {
			 System.out.println(e.getMessage());
		 }
	}
