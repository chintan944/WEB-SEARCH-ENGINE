? prevC++ Program to generate Fibonacci TriangleIn this program, we are getting input from the user for the limit for fibonacci triangle, and printing the fibonacci series for the given number of times (limit).Let's see the C++ example to generate fibonacci triangle.#include <iostream>
using namespace std;
int main()
{
  int a=0,b=1,i,c,n,j;  
    cout<<"Enter the limit: ";  
    cin>>n;  
    for(i=1; i<=n; i++)  
    {  
        a=0;  
        b=1;  
        cout<<b<<"\t"; 
        for(j=1; j<i; j++)  
        {  
            c=a+b;  
          cout<<c<<"\t";  
            a=b;  
            b=c;
        }  
        cout<<"\n";  
    }  
return 0;
}Output:Enter the limit: 10
