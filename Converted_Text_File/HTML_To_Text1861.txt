#include <set>

using namespace std;

int main ()
{
  set<string> myset= {"Java", "C++","SQL"};

  // show content:
  for (auto it = myset.cbegin(); it != myset.cend(); ++it)
    cout <<*it << '\n';
    
  return 0;
}Output:C++
